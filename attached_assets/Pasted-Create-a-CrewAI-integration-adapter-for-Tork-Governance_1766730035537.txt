   class TorkCrewAIMiddleware:
       """Middleware to integrate Tork governance with CrewAI agents."""
       
       def __init__(self, engine: GovernanceEngine = None, agent_id: str = None):
           - Initialize with optional governance engine
           - If no engine provided, create default with PII redaction enabled
           - Store agent_id for tracking

           """Wrap a CrewAI agent with governance controls."""
           - Return a GovernedAgent that intercepts agent actions
       
       def wrap_crew(self, crew) -> "GovernedCrew":
           """Wrap an entire CrewAI crew with governance."""
           - Apply governance to all agents in the crew
       
       def before_task(self, task, agent) -> dict:
           """Called before a task is executed."""
           - Evaluate task input against policies
           - Redact PII if needed
           - Return modified task context or raise GovernanceError
       
       def after_task(self, task, result, agent) -> dict:
           """Called after task completion."""
           - Evaluate output against policies
           - Redact sensitive data from results
           - Generate compliance receipt
           - Return modified result

   class GovernedAgent:
       """A CrewAI agent wrapped with Tork governance."""
       
       def __init__(self, agent, middleware: TorkCrewAIMiddleware):
           self._agent = agent
           self._middleware = middleware
       
       def execute_task(self, task, context=None):
           """Execute task with governance checks."""
           - Call middleware.before_task()
           - Execute original agent task
           - Call middleware.after_task()
           - Return governed result
       
       def __getattr__(self, name):
           """Proxy all other attributes to wrapped agent."""
           return getattr(self._agent, name)
   
   class GovernedCrew:
       """A CrewAI crew with all agents governed."""
       
       def __init__(self, crew, middleware: TorkCrewAIMiddleware):
           self._crew = crew
           self._middleware = middleware
           self._governed_agents = []
       
       def kickoff(self, inputs=None):
           """Run the crew with governance on all tasks."""
           - Apply governance before/after each task
           - Track all compliance receipts
           - Return results with audit trail
