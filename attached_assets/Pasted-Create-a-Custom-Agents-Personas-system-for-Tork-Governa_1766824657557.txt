   class PersonaCapability(str, Enum):
       """Capabilities a persona can have."""
       RESEARCH = "research"
       ANALYSIS = "analysis"
       WRITING = "writing"
       CODING = "coding"
       CRITIQUE = "critique"
       SUMMARIZATION = "summarization"
       TRANSLATION = "translation"
       DATA_PROCESSING = "data_processing"
       CREATIVE = "creative"
       LEGAL = "legal"
       FINANCIAL = "financial"
       MEDICAL = "medical"
   
   class PersonaConfig(BaseModel):
       """Configuration for a custom persona."""
       id: str
       name: str
       description: str = ""
       system_prompt: str  # The core prompt that defines behavior
       capabilities: List[PersonaCapability] = []
       preferred_models: List[str] = []  # ["gpt-4", "claude-3", etc.]
       temperature: float = 0.7
       max_tokens: int = 4096
       
       # Governance settings
       governance_policy: Optional[str] = None  # Custom policy for this persona
       pii_redaction: bool = True
       allowed_actions: List[str] = []  # Actions persona can perform
       blocked_actions: List[str] = []  # Actions persona cannot perform
       max_cost_per_request: float = 1.0
       
       # Metadata
       created_at: datetime = Field(default_factory=datetime.utcnow)
       updated_at: datetime = Field(default_factory=datetime.utcnow)
       created_by: str = ""
       tags: List[str] = []
       metadata: Dict[str, Any] = {}
   
   class PersonaInstance(BaseModel):
       """A running instance of a persona."""
       id: str
       persona_id: str
       session_id: str
       state: Dict[str, Any] = {}  # Persona state/memory
       messages_count: int = 0
       total_tokens: int = 0
       total_cost: float = 0.0
       created_at: datetime = Field(default_factory=datetime.utcnow)
