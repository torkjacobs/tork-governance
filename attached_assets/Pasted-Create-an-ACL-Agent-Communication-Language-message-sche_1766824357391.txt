   class Performative(str, Enum):
       """Standard ACL performatives (speech acts)."""
       REQUEST = "request"          # Ask agent to perform action
       INFORM = "inform"            # Share information
       PROPOSE = "propose"          # Make a proposal
       ACCEPT = "accept"            # Accept a proposal
       REJECT = "reject"            # Reject a proposal
       QUERY = "query"              # Ask for information
       CONFIRM = "confirm"          # Confirm receipt/action
       CANCEL = "cancel"            # Cancel previous request
       FAILURE = "failure"          # Report failure
       AGREE = "agree"              # Agree to perform action
       REFUSE = "refuse"            # Refuse to perform action
       CALL_FOR_PROPOSAL = "cfp"    # Request proposals from agents
   
   class ACLMessage(BaseModel):
       """Standard Agent Communication Language message."""
       id: str = Field(default_factory=lambda: str(uuid4()))
       performative: Performative
       sender: str                  # Agent ID
       receiver: str                # Agent ID or "all" for broadcast
       content: Any                 # Message payload
       reply_to: Optional[str] = None  # ID of message being replied to
       conversation_id: Optional[str] = None  # Thread ID
       protocol: str = "fipa-request"  # Protocol being used
       language: str = "json"       # Content language
       ontology: Optional[str] = None  # Domain ontology
       timestamp: datetime = Field(default_factory=datetime.utcnow)
       expires: Optional[datetime] = None
       metadata: Dict[str, Any] = {}
   
   class Conversation(BaseModel):
       """A conversation thread between agents."""
       id: str
       protocol: str
       participants: List[str]
       messages: List[ACLMessage] = []
       status: str = "active"  # "active", "completed", "failed", "cancelled"
       started_at: datetime = Field(default_factory=datetime.utcnow)
       ended_at: Optional[datetime] = None

3. Create src/tork/acl/protocols.py:
   
   class Protocol(ABC):
       """Base class for interaction protocols."""
       
       name: str
       
       @abstractmethod
       def valid_transitions(self, current_performative: Performative) -> List[Performative]:
           """Return valid next performatives."""
       
       @abstractmethod
       def is_terminal(self, performative: Performative) -> bool:
           """Check if performative ends the protocol."""
   
   class FIPARequestProtocol(Protocol):
       """Standard FIPA Request protocol."""
       name = "fipa-request"
       # request -> agree/refuse -> inform/failure
   
   class FIPAContractNetProtocol(Protocol):
       """FIPA Contract Net protocol for task allocation."""
       name = "fipa-contract-net"
       # cfp -> propose/refuse -> accept/reject -> inform/failure
   
   class FIPAQueryProtocol(Protocol):
       """FIPA Query protocol for information retrieval."""
       name = "fipa-query"
       # query -> inform/failure
