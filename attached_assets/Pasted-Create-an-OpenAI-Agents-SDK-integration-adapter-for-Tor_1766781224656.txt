   class TorkOpenAIAgentsMiddleware:

   class GovernedOpenAIAgent:
       """An OpenAI Agent wrapped with Tork governance."""
       
       def __init__(self, agent, middleware: TorkOpenAIAgentsMiddleware):
           self._agent = agent
           self._middleware = middleware
       
       def run(self, input_text: str, **kwargs):
           """Run agent with governance checks."""
           - Process input through middleware
           - Run original agent

           """Async version of run with governance."""

       def __getattr__(self, name):
           """Proxy all other attributes to wrapped agent."""
